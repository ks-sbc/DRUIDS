    Repository Reference¶

    Overview¶

    DRUIDS consists of 5 main repositories and 3 vault repositories, each serving specific purposes in the infrastructure.

    Repository Setup Commands¶

    Initial Clone¶

    # Create workspace
mkdir -p ~/druids-workspace
cd ~/druids-workspace

# Clone main repositories
git clone https://github.com/ks-sbc/druids-wiki
git clone https://github.com/ks-sbc/.github
git clone https://github.com/ks-sbc/shared
git clone https://github.com/ks-sbc/website

# Clone vault repositories (separate for security)
mkdir -p vault
git clone https://github.com/ks-sbc/vault-public vault/public
git clone https://github.com/ks-sbc/vault-member vault/member
git clone https://github.com/ks-sbc/vault-cadre vault/cadre

    Main Repositories¶

    druids-wiki¶

    • Purpose: Documentation for the DRUIDS system
    • URL: https://github.com/ks-sbc/druids-wiki
    • Structure: Diátaxis framework (tutorials, how-to, reference, explanation)
    • Branch Strategy: main branch with feature/* branches

    Setup Commands¶

    cd druids-wiki
python -m venv venv
source venv/bin/activate
pip install -r requirements.txt

# Development server
mkdocs serve --dev-addr 127.0.0.1:8001

# Build static site
mkdocs build

# Deploy to GitHub Pages
mkdocs gh-deploy

    Key Workflows¶

    # .github/workflows/meeting-minutes.yml
name: Meeting Minutes Generator
on:
  schedule:
    - cron: '0 10 * * 1'  # Every Monday at 10am
  workflow_dispatch:

jobs:
  create-discussion:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      - name: Create Discussion
        uses: abirismyname/create-discussion@v1
        with:
          title: "Meeting Minutes - ${{ env.DATE }}"
          body: |
            # Meeting Minutes
            Date: ${{ env.DATE }}

            ## Agenda
            - [ ] Item 1
            - [ ] Item 2

            ## Attendees
            - [ ] Member 1
            - [ ] Member 2
          category: Meetings

    Symlink Setup Script¶

    #!/bin/bash
# scripts/bash/symlinks.sh

VAULT_PATH=$1
SHARED_PATH="$(cd "$(dirname "${BASH_SOURCE[0]}")/../.." && pwd)"

if [ -z "$VAULT_PATH" ]; then
  echo "Usage: ./symlinks.sh /path/to/vault"
  exit 1
fi

# Create symlinks
ln -sf "$SHARED_PATH/obsidian" "$VAULT_PATH/.obsidian"
ln -sf "$SHARED_PATH/templates" "$VAULT_PATH/_templates"

echo "Symlinks created for $VAULT_PATH"

    • Access: Open to all

    Common Operations¶

    Setting Up a New Vault¶

    # 1. Clone the vault
git clone https://github.com/ks-sbc/vault-new vault/new

# 2. Set up symlinks
cd ~/druids-workspace/shared/scripts/bash
./symlinks.sh ~/druids-workspace/vault/new

# 3. Configure Git
cd ~/druids-workspace/vault/new
git config core.autocrlf false
git config pull.rebase false

# 4. Set up branch protection (via GitHub UI or API)
gh api repos/ks-sbc/vault-new/branches/main/protection \
  --method PUT \
  --field required_status_checks='{"strict":true,"contexts":[]}' \
  --field enforce_admins=false \
  --field required_pull_request_reviews='{"required_approving_review_count":1}' \
  --field restrictions=null

    Submodule Management¶

    # Add website as submodule to public vault
cd vault/public
git submodule add https://github.com/ks-sbc/website website
git submodule update --init --recursive

# Update submodules
git submodule update --remote --merge

# Commit submodule changes
git add website
git commit -m "chore: update website submodule"

    Cross-Repository Operations¶

    # Search across all repositories
fd "*.md" ~/druids-workspace -x rg "search term" {}

# Sync shared configs to all vaults
for vault in public member cadre; do
  cd ~/druids-workspace/vault/$vault
  git pull
  # Symlinks automatically update
done

# Check security classifications
for vault in public member cadre; do
  echo "=== Vault: $vault ==="
  fd "*.md" ~/druids-workspace/vault/$vault -x grep -H "security:" {} | head -5
done

    Security Policies¶

    Branch Protection Rules¶

    # All repositories should have:
main:
  protection_rules:
    - require_pull_request_reviews:
        required_approving_review_count: 1
        dismiss_stale_reviews: true
    - require_status_checks:
        strict: true
        contexts: ["security-audit", "lint"]
    - enforce_admins: false
    - restrictions:
        users: []
        teams: ["ks-sbc/admins"]

    Backup Strategy¶

    #!/bin/bash
# backup-repos.sh

BACKUP_DIR="/secure/backup/druids-$(date +%Y%m%d)"
mkdir -p "$BACKUP_DIR"

# Backup all repositories
for repo in druids-wiki .github shared website vault/public vault/member; do
  echo "Backing up $repo..."
  cd ~/druids-workspace/$repo
  git bundle create "$BACKUP_DIR/${repo//\//-}.bundle" --all
done

# Create encrypted archive
tar -czf - "$BACKUP_DIR" | gpg -c > "$BACKUP_DIR.tar.gz.gpg"
rm -rf "$BACKUP_DIR"

echo "Backup complete: $BACKUP_DIR.tar.gz.gpg"

    Monitoring and Maintenance¶

    Repository Health Check¶

    #!/bin/bash
# check-repos.sh

for repo in druids-wiki .github shared website vault/public vault/member; do
  echo "=== $repo ==="
  cd ~/druids-workspace/$repo

  # Check for uncommitted changes
  if [[ -n $(git status -s) ]]; then
    echo "⚠️  Uncommitted changes found"
  fi

  # Check if behind remote
  git fetch
  BEHIND=$(git rev-list HEAD..@{u} --count)
  if [[ $BEHIND -gt 0 ]]; then
    echo "⚠️  Behind remote by $BEHIND commits"
  fi

  # Check last commit date
  LAST_COMMIT=$(git log -1 --format=%cr)
  echo "Last commit: $LAST_COMMIT"
  echo
done

    Troubleshooting¶

    Common Issues¶

    Symlink Problems¶

    # Check if symlinks are valid
ls -la vault/public/.obsidian
# Should show: .obsidian -> /path/to/shared/obsidian

# Recreate symlinks
rm vault/public/.obsidian
ln -sf ~/druids-workspace/shared/obsidian vault/public/.obsidian

    Permission Errors¶

    # Fix file permissions
find . -type f -name "*.md" -exec chmod 644 {} \;
find . -type d -exec chmod 755 {} \;

# Fix Git permissions
git config core.filemode false

    Submodule Issues¶

    # Reset submodule
git submodule deinit -f website
git submodule update --init

# Clone including submodules
git clone --recurse-submodules https://github.com/ks-sbc/vault-public

    Best Practices¶

    1. Never mix security levels in a single repository
    2. Always use meaningful commit messages
    3. Regularly sync shared configurations
    4. Test changes in one vault before propagating
    5. Document any custom modifications
    6. Backup repositories regularly
    7. Review access permissions quarterly
