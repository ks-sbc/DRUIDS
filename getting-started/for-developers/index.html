    Your Journey: From Setup to Contribution

    This comprehensive guide takes you through the entire DRUIDS technical journey - from initial setup through customization to contributing back to the project. Whether you're implementing DRUIDS for your organization or extending its capabilities, this is your complete reference.

    Prerequisites

    Required Skills

    • Basic Git knowledge (clone, commit, push, pull)
    • Command line comfort (bash/terminal)
    • Markdown familiarity
    • Basic web development (HTML/CSS) helpful but not required

    System Requirements

    • OS: Linux, macOS, or Windows with WSL2
    • Git: Version 2.25+
    • Node.js: 16+ (for development tools)
    • Python: 3.8+ (for MkDocs)
    • Storage: ~2GB for full setup

    Modern CLI Tools (Recommended)

    # Install modern alternatives (much faster!)
cargo install ripgrep fd-find bat exa  # Rust tools
# Or via package manager:
apt install ripgrep fd-find bat exa    # Debian/Ubuntu
brew install ripgrep fd bat exa         # macOS

    Chapter 1: Environment Setup

    1.1 Core Installation

    Clone the Repository Structure

    # Create workspace
mkdir -p ~/druids-workspace/repos
cd ~/druids-workspace

# Clone the three vaults (separate repos)
git clone https://github.com/your-org/vault-public repos/vault/public
git clone https://github.com/your-org/vault-member repos/vault/member
git clone https://github.com/your-org/vault-cadre repos/vault/cadre

# Clone shared configuration
git clone https://github.com/your-org/druids-shared repos/shared

# Clone documentation site
git clone https://github.com/your-org/druids-wiki repos/druids-wiki

    Set Up Symlinks

    # Run the symlink script for each vault
cd repos/shared/scripts/bash
./symlinks.sh ../../vault/public
./symlinks.sh ../../vault/member
# DO NOT symlink L2/cadre - it's FORBIDDEN for AI/LLMs

    1.2 Development Environment

    Install Dependencies

    # Python environment for MkDocs
python -m venv venv

# Node.js for tooling
cd repos/shared
npm install

# Install Obsidian
# Download from https://obsidian.md

    Configure Git

    # Set up user
git config --global user.name "Your Name"
git config --global user.email "your.email@example.com"

# Configure signing (recommended for L1/L2)
git config --global commit.gpgsign true

# Set up credential caching

    2.3 Plugin Architecture

    // Fort Knox plugin structure
class FortKnox {
  constructor() {
    this.securityLevels = ['L0', 'L1', 'L2'];
    this.documentTypes = ['proposal', 'meeting', 'report'];
  }

  validateDocument(file) {
    // Check frontmatter
    const fm = this.getFrontmatter(file);

    // Validate security level
    if (!this.securityLevels.includes(fm.security)) {
      throw new Error(`Invalid security: ${fm.security}`);
    }

    // Check file location matches security
    const path = file.path;
    const expectedSecurity = this.getSecurityFromPath(path);
    if (fm.security !== expectedSecurity) {
      throw new Error('Security mismatch');
    }
  }
}

    3.3 Plugin Development

    Create a Custom Plugin

cd my-plugin
npm install

# Development build
npm run dev

    Example: Auto-Classification Plugin

    import { Plugin, Notice } from 'obsidian';

export default class AutoClassifier extends Plugin {
  async onload() {
    // Listen for file creation
    this.registerEvent(
      this.app.vault.on('create', (file) => {
        this.classifyFile(file);
      })
    );
  }

  async classifyFile(file) {
    const path = file.path;
    let security = 'L0';

    if (path.includes('member')) security = 'L1';
    if (path.includes('cadre')) security = 'L2';

    // Update frontmatter
    await this.app.fileManager.processFrontMatter(
      file,
      (fm) => {
        fm.security = security;
        fm.classified_at = new Date().toISOString();
      }
    );

    new Notice(`Classified as ${security}`);
  }
}

    4.1 GitHub Actions Workflows

    # .github/workflows/security-audit.yml
name: Security Audit

on:
  push:
    branches: [main, L1-internal]
  pull_request:

jobs:
  audit:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3

      - name: Check Security Classifications
        run: |
          # Find misclassified files
          echo "Checking L0 files..."
          find vault/public -name "*.md" -exec grep -L "security: \"L0\"" {} \;

      - name: Validate Cross-Tier Links
        run: |
          # Check for L2 links in public
          ! grep -r "\[\[.*L2.*\]\]" vault/public/

      - name: Document ID Validation
        run: |
          node scripts/validate-document-ids.js

    4.2 Custom Scripts

    Security Scanner

    // scripts/security-scanner.js
const fs = require('fs');
const path = require('path');
const matter = require('gray-matter');

function scanVault(vaultPath, expectedSecurity) {
  const issues = [];

  function scanDir(dir) {
    const files = fs.readdirSync(dir);

    files.forEach(file => {
      const fullPath = path.join(dir, file);
      const stat = fs.statSync(fullPath);

      if (stat.isDirectory() && !file.startsWith('.')) {
        scanDir(fullPath);
      } else if (file.endsWith('.md')) {
        const content = fs.readFileSync(fullPath, 'utf8');
        const { data } = matter(content);

        if (data.security !== expectedSecurity) {
          issues.push({
            file: fullPath,
            expected: expectedSecurity,
            actual: data.security
          });
        }
      }
    });
  }

  scanDir(vaultPath);
  return issues;
}

// Run scan
const publicIssues = scanVault('./vault/public', 'L0');
if (publicIssues.length > 0) {
  console.error('Security violations found:', publicIssues);
  process.exit(1);
}

    4.3 Performance Optimization

    Vault Optimization Script

    #!/bin/bash
# optimize-vault.sh

echo " Optimizing DRUIDS vault..."

# Clean workspace files
find . -name ".obsidian/workspace.json" -delete

# Remove old backups
find . -name "*.bak" -mtime +30 -delete

# Optimize images
find . -name "*.png" -o -name "*.jpg" | while read img; do
  convert "$img" -resize '1920x1920>' -quality 85 "$img"
done

# Clean empty directories
find . -type d -empty -delete

# Generate size report
echo " Vault size report:"
du -sh vault/*

    Chapter 5: Testing & Quality

    5.1 Testing Framework

    // tests/security.test.js
const { validateSecurity } = require('../lib/security');

describe('Security Validation', () => {
  test('validates correct security levels', () => {
    expect(validateSecurity('L0')).toBe(true);
    expect(validateSecurity('L1')).toBe(true);
    expect(validateSecurity('L2')).toBe(true);
    expect(validateSecurity('L3')).toBe(false);
  });

  test('enforces path-security alignment', () => {
    expect(validatePath('public/doc.md', 'L0')).toBe(true);
    expect(validatePath('public/doc.md', 'L1')).toBe(false);
  });
});

    5.2 Pre-commit Hooks

    # .git/hooks/pre-commit
#!/bin/bash

echo "Running pre-commit checks..."

# Check for secrets
if git diff --cached --name-only | xargs grep -E "(password|secret|key)"; then
  echo " Potential secrets detected!"
  exit 1
fi

# Validate markdown
git diff --cached --name-only --diff-filter=ACM | grep "\.md$" | while read file; do
  markdownlint "$file" || exit 1
done

# Run security audit
npm run security-audit || exit 1

echo " All checks passed!"

            --exclude='.git' \
            --exclude='node_modules' \
            vault/public

      - name: Upload artifacts
        uses: actions/upload-release-asset@v1
        with:
          upload_url: ${{ github.event.release.upload_url }}
          asset_path: ./druids-public-${{ github.ref_name }}.tar.gz

    7.1 Contribution Guidelines

    Code Standards

    //  Good: Clear, documented, secure
export function generateDocumentId(type, security) {
  if (!VALID_TYPES.includes(type)) {
    throw new Error(`Invalid type: ${type}`);
  }

  const prefix = TYPE_PREFIXES[type];
  const date = new Date();
  const sequence = getNextSequence(type);

  return `${prefix}-${date.getFullYear()}-${sequence}-${security}`;
}

//  Bad: Unclear, undocumented, insecure
export function genId(t, s) {
  return t + '-' + Date.now() + '-' + s;
}
